```
✅ Password protection: No hardcoded credentials
✅ Code review: No secrets in git history
✅ Logging: Audit who accessed trade secrets
✅ Incident response: Plan for theft/breach
```

#### Implement License Key System:

```python
# backend/licensing.py

import os
import hmac
import hashlib
from datetime import datetime, timedelta

class LicenseValidator:
    """Validate that WAI instance has valid license"""
    
    def __init__(self):
        self.master_key = os.environ.get("LICENSE_MASTER_KEY")
        if not self.master_key:
            raise RuntimeError("FATAL: LICENSE_MASTER_KEY not set")
    
    def generate_license(self, customer_id: str, duration_days: int = 365) -> str:
        """
        Generate a license key for a customer.
        Only you (headquarters) can generate these.
        """
        expires = datetime.utcnow() + timedelta(days=duration_days)
        
        # License format: CUSTOMER_ID|EXPIRY_DATE|SIGNATURE
        license_string = f"{customer_id}|{expires.isoformat()}"
        
        # Sign with master key
        signature = hmac.new(
            self.master_key.encode(),
            license_string.encode(),
            hashlib.sha256
        ).hexdigest()
        
        license_key = f"{license_string}|{signature}"
        return license_key
    
    def validate_license(self, license_key: str) -> tuple[bool, str]:
        """
        Validate license on startup.
        If invalid, platform refuses to start.
        """
        try:
            parts = license_key.split("|")
            if len(parts) != 3:
                return False, "Invalid license format"
            
            customer_id, expires_str, signature = parts
            
            # Verify signature
            expected_sig = hmac.new(
                self.master_key.encode(),
                f"{customer_id}|{expires_str}".encode(),
                hashlib.sha256
            ).hexdigest()
            
            if not hmac.compare_digest(signature, expected_sig):
                return False, "License signature invalid (tampered)"
            
            # Check expiration
            expires = datetime.fromisoformat(expires_str)
            if datetime.utcnow() > expires:
                return False, f"License expired on {expires.isoformat()}"
            
            return True, "License valid"
        
        except Exception as e:
            return False, f"License validation error: {e}"

# Usage in server startup:
validator = LicenseValidator()
is_valid, message = validator.validate_license(os.environ.get("WAI_LICENSE"))

if not is_valid:
    logger.critical(f"FATAL: {message}")
    raise RuntimeError(f"Platform cannot start: {message}")

logger.info(f"✅ License valid: {message}")
```

#### Distribution Model:

```
1. Customer requests WAI platform
2. You (HQ) generate license key:
   LICENSE_KEY = validator.generate_license("acme-corp", duration_days=365)
3. Send license via secure channel (encrypted email)
```

```python
# backend/instance.py

import os
import uuid
from datetime import datetime

class InstanceFingerprint:
    """
    Each WAI deployment gets unique fingerprint.
    Used to prove which instance data came from.
    """
    
    @staticmethod
    def generate_fingerprint() -> str:
        """Generate unique fingerprint for this deployment"""
        instance_id = str(uuid.uuid4())
        deployment_date = datetime.utcnow().isoformat()
        
        # Store in environment
        os.environ["INSTANCE_ID"] = instance_id
        os.environ["DEPLOYED_AT"] = deployment_date
        
        return instance_id
    
    @staticmethod
    def get_fingerprint() -> str:
        """Get existing fingerprint (or generate if missing)"""
        instance_id = os.environ.get("INSTANCE_ID")
        
        if not instance_id:
            instance_id = InstanceFingerprint.generate_fingerprint()
        
        return instance_id

# On startup:

logger.info(f"Instance fingerprint: {fingerprint}")
```

```python
# backend/api_keys.py

from fastapi import HTTPException, Header
from datetime import datetime

class APIKeyValidator:
    """
    All API access requires valid API key.
    Prevents unauthorized copying/scraping.
    """
    
    async def validate_key(self, api_key: str = Header(None)) -> dict:
        """Validate API key from header"""
        if not api_key:
            raise HTTPException(400, "Missing API key (Authorization header)")
        
        # Look up key in database
        key_record = await db.api_keys.find_one({"key": api_key})
        
        if not key_record:
            raise HTTPException(401, "Invalid API key")
        
        if key_record.get("is_active") is False:
            raise HTTPException(401, "API key is disabled")
        
        # Check rate limit
        rate_limit = key_record.get("rate_limit", 100)  # 100 req/min
        current_requests = await redis.get(f"api_key_requests:{api_key}")
        
        if current_requests and int(current_requests) >= rate_limit:
            raise HTTPException(429, "Rate limit exceeded")
        
        # Log request
        await redis.incr(f"api_key_requests:{api_key}")
        await redis.expire(f"api_key_requests:{api_key}", 60)
        
        await db.api_key_usage.insert_one({
            "api_key_id": key_record["_id"],
            "timestamp": datetime.utcnow(),
            "endpoint": request.url.path,
            "method": request.method,
        })
        
        return key_record
```
